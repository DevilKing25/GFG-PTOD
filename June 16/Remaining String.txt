class Solution {
    public String printString(String s, char ch, int count) {
        // code here
        for(int i=0; i<s.length();i++){
            if(s.charAt(i)==Character.toUpperCase(ch) || s.charAt(i) == Character.toLowerCase(ch)){
                count=count-1;
            }
            if(count==0){
                return s.substring(i+1);
        }
    }
    return "";
    }
}
